  <script>
    // --- DATA ---
    const greetings = ["Hai", "Halo", "Ciao", "Hola", "Ola", "Hi!", "Guten", "Salut"];
    const codes = ["< />", "var", "{ }", "404", "if()", "npm", "git", "src", "dev", "cmd"];

    // --- LOGIKA HEADER (HOVER) ---
    const headerLogo = document.getElementById('headerLogo');
    const headerText = headerLogo.querySelector('#text-content');

    headerLogo.addEventListener('mouseenter', () => {
      // 1. Ambil kata acak
      const randomGreet = greetings[Math.floor(Math.random() * greetings.length)];
      // 2. Masukkan ke SVG
      headerText.textContent = randomGreet;
      // 3. Tambah class untuk trigger CSS transition
      headerLogo.classList.add('show-text');
    });

    headerLogo.addEventListener('mouseleave', () => {
      // Hapus class, kembali ke ikon kurung
      headerLogo.classList.remove('show-text');
    });


    // --- LOGIKA LOADING (TYPING CEPAT) ---
    // (Setup: Copy SVG ke div loading dulu)
    const loadingLogo = document.getElementById('loadingLogo');
    loadingLogo.innerHTML = headerLogo.innerHTML; // Copy SVG dari atas
    
    const loadingText = loadingLogo.querySelector('#text-content');
    loadingLogo.classList.add('show-text'); // Paksa mode teks selalu aktif

    let index = 0;
    setInterval(() => {
      // Ganti teks setiap 300ms
      loadingText.textContent = codes[index];
      index = (index + 1) % codes.length;
    }, 300); // Kecepatan ketikan
  </script>
